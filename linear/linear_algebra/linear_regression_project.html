def matxMultiply(A, B):
    _,n = shape(A)
    m,_ = shape(B)
    if not n == m:
        raise ValueError("Matrix A\'s column number doesn\'t equal to Matrix b\'s row number")
    else:
        return [ [sum( a * b for a, b in zip(a,b)) for b in zip(*B)] for a in A ]

def gj_Solve(A, b, decPts=4, epsilon=1.0e-16):
    
    m,n = shape(A)
    if not m==n:
        raise ValueError('A')
    if len(A) != len(b):
        return None
    
    Ab = augmentMatrix(A,b)
    for col in range(m):  
        
##########################################################################
#
#         #
#         max_value_pre_row_under_line = {}
#         for row in range(col,m)[::-1]:
#             max_value_pre_row_under_line[max([abs(x) for x in Ab[row][0:row+1]])] = row
#         max_val = max(max_value_pre_row_under_line.keys())
#         #
#         if abs(max_val) < epsilon: 
#             return None 
#         swaped_row = max_value_pre_row_under_line[max_val]
##########################################################################

        # 
        maxs = transpose(Ab)[col][col:] 
        max_val = max(maxs, key=abs)
        
        #
        if abs(max_val) < epsilon: 
            return None
        
        #
        max_idx = maxs.index(max_val) + col
        swapRows(Ab, col, max_idx)
        
        #1
        scale = 1/Ab[col][col]
        scaleRow(Ab, col, scale)
            
        #col
        for row in range(m):
            if not row==col and not Ab[row][col]==0:
                addScaledRow(Ab, row, col, -Ab[row][col])
        
    result = [[row[-1]] for row in Ab ]
    return result

def calculateMSE2D(X,Y,m,b):
    sqrt_sum = sum([(y - m * x - b) ** 2 for x,y in zip(X,Y)])
